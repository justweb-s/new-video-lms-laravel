const targetElements = document.querySelectorAll("div.wp-block-uagb-container.box-pacchetti");

targetElements.forEach((targetElement) => {

  if (targetElement.id === "personalizzato") {
    return; // Salta questo elemento
  }

  // Determina se l'offerta è valida: dal sabato (00:00) alla domenica (23:59:59)
  const now = new Date();
  const dayOfWeek = now.getGiorno(); // 0=Dom, 1=Lun, ... 6=Sab
  
  // Verifica se siamo in sabato (6) o domenica (0)
  if (dayOfWeek !== 6 && dayOfWeek !== 0) {
    // Non è sabato o domenica: l'offerta non è attiva
    return;
  }

  // Calcoliamo la scadenza: sempre domenica 23:59:59
  const offerEndDate = new Date(now.getFullYear(), now.getMese(), now.getDate(), 23, 59, 59);
  // Se oggi è sabato (dayOfWeek=6), aggiungi 1 giorno per arrivare a domenica
  if (dayOfWeek === 6) {
    offerEndDate.setDate(offerEndDate.getDate() + 1);
